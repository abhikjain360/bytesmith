impl<'a> ::binparse::Dissect<'a> for Opt<'a> {
    fn field_tree(&mut self) -> ::binparse::FieldNode<'a> {
        Opt::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::binparse::Handoff<'a>> {
        Opt::handoff(self)
    }
}

impl<'a> Opt<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::binparse::ParseError> {
        let mut me = Self { data };
        me.flags_fatal_check()?;
        me.flags_recoverable_check()?;
        if me.conditional_0_present() {
            me.a_fatal_check()?;
            me.a_recoverable_check()?;
        }
        if me.conditional_0_present() {
            me.b_fatal_check()?;
            me.b_recoverable_check()?;
        }
        let len = me.conditional_0_end_offset();
        if len.bit != 0 {
            return Err(::binparse::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::binparse::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::binparse::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.flags_present_node());
        }
        {
            let me = &mut *self;
            if me.conditional_0_present() {
                children.push(me.a_present_node());
            } else {
                children.push(me.a_absent_node());
            }
        }
        {
            let me = &mut *self;
            if me.conditional_0_present() {
                children.push(me.b_present_node());
            } else {
                children.push(me.b_absent_node());
            }
        }
        let mut root = ::binparse::FieldNode::new(
                "Opt",
                "Opt",
                0usize..self.conditional_0_end_offset().bits(),
                ::binparse::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::binparse::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::binparse::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::binparse::ParseError> = None;
        if fatal.is_none() {
            match me.flags_fatal_check() {
                Err(error) => {
                    let start = me.flags_start_offset().bits();
                    children
                        .push(
                            ::binparse::FieldNode::new(
                                    "flags",
                                    "u8",
                                    start..start,
                                    ::binparse::Value::Opaque,
                                )
                                .with_status(::binparse::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.flags_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.flags_bit_range();
                            children
                                .push(
                                    ::binparse::FieldNode::new(
                                            "flags",
                                            "u8",
                                            bit_range,
                                            ::binparse::Value::Opaque,
                                        )
                                        .with_status(::binparse::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.flags_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            if me.conditional_0_present() {
                match me.a_fatal_check() {
                    Err(error) => {
                        let start = me.a_start_offset().bits();
                        children
                            .push(
                                ::binparse::FieldNode::new(
                                        "a",
                                        "u16",
                                        start..start,
                                        ::binparse::Value::Opaque,
                                    )
                                    .with_status(::binparse::Status::Error(error)),
                            );
                        fatal = Some(error);
                    }
                    Ok(()) => {
                        match me.a_recoverable_check() {
                            Err(error) => {
                                let bit_range = me.a_bit_range();
                                children
                                    .push(
                                        ::binparse::FieldNode::new(
                                                "a",
                                                "u16",
                                                bit_range,
                                                ::binparse::Value::Opaque,
                                            )
                                            .with_status(::binparse::Status::Error(error)),
                                    );
                            }
                            Ok(()) => {
                                children.push(me.a_present_node());
                            }
                        }
                    }
                }
            } else {
                children.push(me.a_absent_node());
            }
        }
        if fatal.is_none() {
            if me.conditional_0_present() {
                match me.b_fatal_check() {
                    Err(error) => {
                        let start = me.b_start_offset().bits();
                        children
                            .push(
                                ::binparse::FieldNode::new(
                                        "b",
                                        "u8",
                                        start..start,
                                        ::binparse::Value::Opaque,
                                    )
                                    .with_status(::binparse::Status::Error(error)),
                            );
                        fatal = Some(error);
                    }
                    Ok(()) => {
                        match me.b_recoverable_check() {
                            Err(error) => {
                                let bit_range = me.b_bit_range();
                                children
                                    .push(
                                        ::binparse::FieldNode::new(
                                                "b",
                                                "u8",
                                                bit_range,
                                                ::binparse::Value::Opaque,
                                            )
                                            .with_status(::binparse::Status::Error(error)),
                                    );
                            }
                            Ok(()) => {
                                children.push(me.b_present_node());
                            }
                        }
                    }
                }
            } else {
                children.push(me.b_absent_node());
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::binparse::FieldNode::new(
                "Opt",
                "Opt",
                0usize..root_end,
                ::binparse::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::binparse::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::binparse::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn flags_present_node(&mut self) -> ::binparse::FieldNode<'a> {
        let bit_range = self.flags_bit_range();
        ::binparse::FieldNode::new(
            "flags",
            "u8",
            bit_range.clone(),
            ::binparse::Value::UInt(u128::from(self.flags())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn flags(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn flags_end_offset(&mut self) -> binparse::Len {
        binparse::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn flags_start_offset(&mut self) -> binparse::Len {
        binparse::Len::ZERO
    }
    pub fn flags_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.flags_start_offset().bits()..self.flags_end_offset().bits()
    }
    #[allow(dead_code)]
    fn flags_fatal_check(&mut self) -> Result<(), ::binparse::ParseError> {
        {
            let len = self.flags_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::binparse::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn flags_recoverable_check(&mut self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
    #[allow(dead_code, unused_parens)]
    fn conditional_0_present(&mut self) -> bool {
        ((self.flags() as usize) > (0usize))
    }
    #[allow(dead_code)]
    fn a_present_node(&mut self) -> ::binparse::FieldNode<'a> {
        let bit_range = self.a_bit_range();
        ::binparse::FieldNode::new(
            "a",
            "u16",
            bit_range.clone(),
            ::binparse::Value::UInt(u128::from(self.a_raw())),
        )
    }
    #[allow(dead_code)]
    fn a_absent_node(&mut self) -> ::binparse::FieldNode<'a> {
        let start = self.a_start_offset().bits();
        ::binparse::FieldNode::new("a", "u16", start..start, ::binparse::Value::Absent)
            .hide()
    }
    #[allow(clippy::identity_op)]
    fn a_raw(&mut self) -> u16 {
        u16::from_be_bytes(self.data[1usize..3usize].try_into().unwrap())
    }
    pub fn a(&mut self) -> Option<u16> {
        if self.conditional_0_present() { Some(self.a_raw()) } else { None }
    }
    pub fn a_end_offset(&mut self) -> binparse::Len {
        binparse::Len {
            byte: 3usize,
            bit: 0usize,
        }
    }
    pub fn a_start_offset(&mut self) -> binparse::Len {
        self.flags_end_offset()
    }
    pub fn a_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.a_start_offset().bits()..self.a_end_offset().bits()
    }
    #[allow(dead_code)]
    fn a_fatal_check(&mut self) -> Result<(), ::binparse::ParseError> {
        {
            let len = self.a_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::binparse::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn a_recoverable_check(&mut self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn b_present_node(&mut self) -> ::binparse::FieldNode<'a> {
        let bit_range = self.b_bit_range();
        ::binparse::FieldNode::new(
            "b",
            "u8",
            bit_range.clone(),
            ::binparse::Value::UInt(u128::from(self.b_raw())),
        )
    }
    #[allow(dead_code)]
    fn b_absent_node(&mut self) -> ::binparse::FieldNode<'a> {
        let start = self.b_start_offset().bits();
        ::binparse::FieldNode::new("b", "u8", start..start, ::binparse::Value::Absent)
            .hide()
    }
    #[allow(clippy::identity_op)]
    fn b_raw(&mut self) -> u8 {
        self.data[3usize]
    }
    pub fn b(&mut self) -> Option<u8> {
        if self.conditional_0_present() { Some(self.b_raw()) } else { None }
    }
    pub fn b_end_offset(&mut self) -> binparse::Len {
        binparse::Len {
            byte: 4usize,
            bit: 0usize,
        }
    }
    pub fn b_start_offset(&mut self) -> binparse::Len {
        self.a_end_offset()
    }
    pub fn b_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.b_start_offset().bits()..self.b_end_offset().bits()
    }
    #[allow(dead_code)]
    fn b_fatal_check(&mut self) -> Result<(), ::binparse::ParseError> {
        {
            let len = self.b_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::binparse::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn b_recoverable_check(&mut self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
    fn conditional_0_end_offset(&mut self) -> binparse::Len {
        if self.conditional_0_present() {
            self.b_end_offset()
        } else {
            self.flags_end_offset()
        }
    }
}

impl<'a> OptWriter<'a> {
    #[allow(unused_parens)]
    pub fn encoded_len(content: &OptContent) -> usize {
        1usize + if (((content.flags as usize)) > (0usize)) { 3usize } else { 0usize }
    }
    pub fn set_flags(&mut self, value: u8) -> &mut Self {
        self.data[0usize] = value;
        self
    }
    #[allow(unused_parens)]
    pub fn write_into(
        data: &'a mut [u8],
        content: &OptContent,
    ) -> ::binparse::WriteResult<usize> {
        let need = Self::encoded_len(content);
        if data.len() < need {
            return Err(::binparse::WriteError::NotEnoughSpace {
                expected: need,
                got: data.len(),
            });
        }
        let mut w = Self { data };
        w.set_flags(content.flags);
        if (((content.flags as usize)) > (0usize)) {
            if let ::core::option::Option::Some(value) = content.a {
                let value: u16 = value;
                let branch = &mut w.data[1usize..];
                branch[0usize..2usize].copy_from_slice(&value.to_be_bytes());
            }
            if let ::core::option::Option::Some(value) = content.b {
                let value: u8 = value;
                let branch = &mut w.data[1usize..];
                branch[2usize] = value;
            }
        }
        Ok(need)
    }
    pub fn to_vec(content: &OptContent) -> ::std::vec::Vec<u8> {
        let mut buf = ::std::vec![0u8; Self::encoded_len(content)];
        let _ = OptWriter::write_into(&mut buf, content);
        buf
    }
}

pub struct Opt<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}

pub struct OptContent {
    pub flags: u8,
    pub a: ::core::option::Option<u16>,
    pub b: ::core::option::Option<u8>,
}

pub struct OptWriter<'a> {
    data: &'a mut [u8],
}
