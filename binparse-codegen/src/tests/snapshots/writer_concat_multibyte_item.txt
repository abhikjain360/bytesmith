impl<'a> MixWriter<'a> {
    pub const SIZE: usize = 6usize;
    pub fn new(data: &'a mut [u8]) -> ::binparse::WriteResult<Self> {
        if data.len() < Self::SIZE {
            return Err(::binparse::WriteError::NotEnoughSpace {
                expected: Self::SIZE,
                got: data.len(),
            });
        }
        Ok(Self { data })
    }
    pub fn writer_over(data: &'a mut [u8]) -> ::binparse::WriteResult<Self> {
        if data.len() < Self::SIZE {
            return Err(::binparse::WriteError::NotEnoughSpace {
                expected: Self::SIZE,
                got: data.len(),
            });
        }
        Ok(Self { data })
    }
    pub fn set_tag(&mut self, value: u8) -> &mut Self {
        self.data[0usize] = value;
        self
    }
    pub fn set_combo(&mut self, value: ([u8; 2usize], u16)) -> &mut Self {
        self.data[1usize..3usize].copy_from_slice(&value.0);
        {
            let value = value.1;
            self.data[3usize..5usize].copy_from_slice(&value.to_be_bytes());
        }
        self
    }
    pub fn set_trailer(&mut self, value: u8) -> &mut Self {
        self.data[5usize] = value;
        self
    }
    pub fn write_into(
        data: &'a mut [u8],
        content: &MixContent,
    ) -> ::binparse::WriteResult<usize> {
        let mut w = Self::new(data)?;
        w.set_tag(content.tag);
        w.set_combo(content.combo);
        w.set_trailer(content.trailer);
        Ok(Self::SIZE)
    }
    pub fn to_vec(content: &MixContent) -> ::std::vec::Vec<u8> {
        let mut buf = ::std::vec![0u8; Self::SIZE];
        let _ = MixWriter::write_into(&mut buf, content);
        buf
    }
}
