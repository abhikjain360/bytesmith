    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::binparse::ParseError> {
        let me = Self { data };
        me.dst_fatal_check()?;
        me.dst_recoverable_check()?;
        me.src_fatal_check()?;
        me.src_recoverable_check()?;
        me.tpid_fatal_check()?;
        me.tpid_recoverable_check()?;
        me.pcp_fatal_check()?;
        me.pcp_recoverable_check()?;
        me.dei_fatal_check()?;
        me.dei_recoverable_check()?;
        me.vid_hi_fatal_check()?;
        me.vid_hi_recoverable_check()?;
        me.vid_lo_fatal_check()?;
        me.vid_lo_recoverable_check()?;
        me.ethertype_fatal_check()?;
        me.ethertype_recoverable_check()?;
        me.payload_fatal_check()?;
        me.payload_recoverable_check()?;
        let len = me.payload_end_offset();
        if len.bit != 0 {
            return Err(::binparse::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::binparse::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }

    #[allow(dead_code)]
    fn dst_present_node(&self) -> ::binparse::FieldNode<'a> {
        let bit_range = self.dst_bit_range();
        {
            let mut elem_nodes = ::std::vec::Vec::new();
            if let Ok(iter) = self.dst() {
                let mut start = bit_range.start;
                for (i, elem) in iter.enumerate() {
                    let end = start.saturating_add(8usize);
                    match elem {
                        Ok(value) => {
                            elem_nodes
                                .push(
                                    ::binparse::FieldNode::new(
                                        i.to_string(),
                                        "u8",
                                        start..end,
                                        ::binparse::Value::UInt(u128::from(value)),
                                    ),
                                )
                        }
                        Err(error) => {
                            elem_nodes
                                .push(
                                    ::binparse::FieldNode::new(
                                            i.to_string(),
                                            "u8",
                                            start..start,
                                            ::binparse::Value::Opaque,
                                        )
                                        .with_status(::binparse::Status::Error(error)),
                                );
                        }
                    }
                    start = end;
                }
            }
            ::binparse::FieldNode::new(
                    "dst",
                    "[u8]",
                    bit_range.clone(),
                    ::binparse::Value::Array,
                )
                .with_children(elem_nodes)
        }
    }

            if self.data.len() < expected {
                return Err(::binparse::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn dei_recoverable_check(&self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn vid_hi_present_node(&self) -> ::binparse::FieldNode<'a> {
        let bit_range = self.vid_hi_bit_range();
        ::binparse::FieldNode::new(
            "vid_hi",
            "b<4>",
            bit_range.clone(),
            ::binparse::Value::UInt(u128::from(self.vid_hi())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn vid_hi(&self) -> u8 {
        (self.data[14usize] >> 0usize) & 15u8
    }
    pub fn vid_hi_end_offset(&self) -> binparse::Len {
        binparse::Len {
            byte: 15usize,
            bit: 0usize,
        }
    }
    pub fn vid_hi_start_offset(&self) -> binparse::Len {
        self.dei_end_offset()
    }
    pub fn vid_hi_bit_range(&self) -> ::core::ops::Range<usize> {
        self.vid_hi_start_offset().bits()..self.vid_hi_end_offset().bits()
    }
    #[allow(dead_code)]

        }

    fn ethertype_recoverable_check(&self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn payload_present_node(&self) -> ::binparse::FieldNode<'a> {
        let bit_range = self.payload_bit_range();
        {
            let mut elem_nodes = ::std::vec::Vec::new();
            if let Ok(iter) = self.payload() {
                let mut start = bit_range.start;
                for (i, elem) in iter.enumerate() {
                    let end = start.saturating_add(8usize);
                    match elem {
                        Ok(value) => {
                            elem_nodes
                                .push(
                                    ::binparse::FieldNode::new(
                                        i.to_string(),
                                        "u8",
                                        start..end,
                                        ::binparse::Value::UInt(u128::from(value)),
                                    ),
                                )
                        }
                        Err(error) => {
                            elem_nodes
                                .push(
                                    ::binparse::FieldNode::new(
                                            i.to_string(),
                                            "u8",
                                            start..start,
                                            ::binparse::Value::Opaque,
                                        )
                                        .with_status(::binparse::Status::Error(error)),
                                );
                        }
                    }
                    start = end;
                }
            }
            ::binparse::FieldNode::new(
                    "payload",
                    "[u8]",
                    bit_range.clone(),
                    ::binparse::Value::Array,
                )
                .with_children(elem_nodes)
        }
    }
    #[allow(clippy::identity_op)]
    pub fn payload(&self) -> ::binparse::ParseResult<Vlan_payload_Iterator<'a>> {
        Ok(Vlan_payload_Iterator {
            idx: 0,
            count: self.data.len().saturating_sub(18usize),
            data: &self.data[18usize..],
        })
    }
    pub fn payload_end_offset(&self) -> binparse::Len {
        ::binparse::Len {
            byte: 18usize,
            bit: 0usize,
        }
            + ({
                {
                    let start = 18usize;
                    ::binparse::Len {
                        byte: self.data.len().saturating_sub(start),
                        bit: 0,
                    }
                }
            })
    }
    pub fn payload_start_offset(&self) -> binparse::Len {
        self.ethertype_end_offset()
    }
    pub fn payload_bit_range(&self) -> ::core::ops::Range<usize> {
        self.payload_start_offset().bits()..self.payload_end_offset().bits()
    }
    #[allow(dead_code)]
    fn payload_fatal_check(&self) -> Result<(), ::binparse::ParseError> {
        {
            let len = self.payload_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::binparse::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn payload_recoverable_check(&self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
}

impl<'a> VlanWriter<'a> {
    pub fn encoded_len(lens: &VlanLens) -> usize {
        18usize + lens.payload
    }
    pub fn new(data: &'a mut [u8], lens: VlanLens) -> ::binparse::WriteResult<Self> {
        let need = Self::encoded_len(&lens);
        if data.len() < need {
            return Err(::binparse::WriteError::NotEnoughSpace {
                expected: need,
                got: data.len(),
            });
        }
        Ok(Self { data, lens })
    }
    pub fn writer_over(data: &'a mut [u8]) -> ::binparse::WriteResult<Self> {
        let lens = {
            let (view, _) = Vlan::parse(data)
                .map_err(|_| ::binparse::WriteError::InvalidContent)?;
            let region_bytes = view.payload_end_offset().byte
                - view.payload_start_offset().byte;
            VlanLens { payload: region_bytes }
        };
        let need = Self::encoded_len(&lens);
        if data.len() < need {
            return Err(::binparse::WriteError::NotEnoughSpace {
                expected: need,
                got: data.len(),
            });
        }
        Ok(Self { data, lens })
    }
    pub fn dst_mut(&mut self) -> &mut [u8] {
        &mut self.data[0usize..6usize]
    }
    pub fn src_mut(&mut self) -> &mut [u8] {
        &mut self.data[6usize..12usize]
    }
    pub fn set_pcp(&mut self, value: u8) -> &mut Self {
        self.data[14usize] = (self.data[14usize] & 31u8) | ((value & 7u8) << 5usize);
        self
    }
    pub fn set_dei(&mut self, value: u8) -> &mut Self {
        self.data[14usize] = (self.data[14usize] & 239u8) | ((value & 1u8) << 4usize);
        self
    }
    pub fn set_vid_hi(&mut self, value: u8) -> &mut Self {
        self.data[14usize] = (self.data[14usize] & 240u8) | ((value & 15u8) << 0usize);
        self
    }
    pub fn set_vid_lo(&mut self, value: u8) -> &mut Self {
        self.data[15usize] = value;
        self
    }
    pub fn set_ethertype(&mut self, value: u16) -> &mut Self {
        self.data[16usize..18usize].copy_from_slice(&value.to_be_bytes());
        self
    }
    pub fn payload_mut(&mut self) -> &mut [u8] {
        let off = 18usize;
        &mut self.data[off..off + self.lens.payload]
    }
    pub fn write_into(
        data: &'a mut [u8],
        content: &VlanContent,
    ) -> ::binparse::WriteResult<usize> {
        let lens = VlanLens {
            payload: content.payload.len(),
        };
        let need = Self::encoded_len(&lens);
        let mut w = Self::new(data, lens)?;
        w.dst_mut().copy_from_slice(&content.dst);
        w.src_mut().copy_from_slice(&content.src);
        {
            let value: u16 = 33024;
            w.data[12usize..14usize].copy_from_slice(&value.to_be_bytes());
        }
        w.set_pcp(content.pcp);
        w.set_dei(content.dei);
        w.set_vid_hi(content.vid_hi);
        w.set_vid_lo(content.vid_lo);
        w.set_ethertype(content.ethertype);
        w.payload_mut().copy_from_slice(content.payload);
        Ok(need)
    }
    pub fn to_vec(content: &VlanContent) -> ::std::vec::Vec<u8> {
        let lens = VlanLens {
            payload: content.payload.len(),
        };
        let mut buf = ::std::vec![0u8; Self::encoded_len(& lens)];
        let _ = VlanWriter::write_into(&mut buf, content);
        buf
    }
}

pub struct Vlan<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}

pub struct VlanContent<'a> {
    pub dst: [u8; 6usize],
    pub src: [u8; 6usize],
    pub pcp: u8,
    pub dei: u8,
    pub vid_hi: u8,
    pub vid_lo: u8,
    pub ethertype: u16,
    pub payload: &'a [u8],
}

pub struct VlanWriter<'a> {
    data: &'a mut [u8],
    lens: VlanLens,
}
