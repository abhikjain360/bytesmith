#[allow(non_camel_case_types)]
pub struct Rec_items_Iterator<'a> {
    idx: usize,
    count: usize,
    data: &'a [u8],
}

#[derive(Clone, Copy)]
pub struct RecLens {
    pub items: usize,
}

impl<'a> ::binparse::Dissect<'a> for Pair<'a> {
    fn field_tree(&mut self) -> ::binparse::FieldNode<'a> {
        Pair::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::binparse::Handoff<'a>> {
        Pair::handoff(self)
    }
}

impl<'a> ::binparse::Dissect<'a> for Rec<'a> {
    fn field_tree(&mut self) -> ::binparse::FieldNode<'a> {
        Rec::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::binparse::Handoff<'a>> {
        Rec::handoff(self)
    }
}

impl<'a> ::std::iter::Iterator for Rec_items_Iterator<'a> {
    type Item = ::binparse::ParseResult<Pair<'a>>;
    fn next(&mut self) -> std::option::Option<Self::Item> {
        if self.idx == self.count {
            return None;
        }
        self.idx += 1;
        match Pair::parse(self.data) {
            Ok((value, rem)) => {
                self.data = rem;
                Some(Ok(value))
            }
            Err(error) => Some(Err(error)),
        }
    }
}

impl<'a> Pair<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::binparse::ParseError> {
        let mut me = Self { data };
        me.a_fatal_check()?;
        me.a_recoverable_check()?;
        me.b_fatal_check()?;
        me.b_recoverable_check()?;
        let len = me.b_end_offset();
        if len.bit != 0 {
            return Err(::binparse::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::binparse::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::binparse::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.a_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.b_present_node());
        }
        let mut root = ::binparse::FieldNode::new(
                "Pair",
                "Pair",
                0usize..self.b_end_offset().bits(),
                ::binparse::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::binparse::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::binparse::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::binparse::ParseError> = None;
        if fatal.is_none() {
            match me.a_fatal_check() {
                Err(error) => {
                    let start = me.a_start_offset().bits();
                    children
                        .push(
                            ::binparse::FieldNode::new(
                                    "a",
                                    "u8",
                                    start..start,
                                    ::binparse::Value::Opaque,
                                )
                                .with_status(::binparse::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.a_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.a_bit_range();
                            children
                                .push(
                                    ::binparse::FieldNode::new(
                                            "a",
                                            "u8",
                                            bit_range,
                                            ::binparse::Value::Opaque,
                                        )
                                        .with_status(::binparse::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.a_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.b_fatal_check() {
                Err(error) => {
                    let start = me.b_start_offset().bits();
                    children
                        .push(
                            ::binparse::FieldNode::new(
                                    "b",
                                    "u16",
                                    start..start,
                                    ::binparse::Value::Opaque,
                                )
                                .with_status(::binparse::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.b_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.b_bit_range();
                            children
                                .push(
                                    ::binparse::FieldNode::new(
                                            "b",
                                            "u16",
                                            bit_range,
                                            ::binparse::Value::Opaque,
                                        )
                                        .with_status(::binparse::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.b_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::binparse::FieldNode::new(
                "Pair",
                "Pair",
                0usize..root_end,
                ::binparse::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::binparse::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::binparse::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn a_present_node(&mut self) -> ::binparse::FieldNode<'a> {
        let bit_range = self.a_bit_range();
        ::binparse::FieldNode::new(
            "a",
            "u8",
            bit_range.clone(),
            ::binparse::Value::UInt(u128::from(self.a())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn a(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn a_end_offset(&mut self) -> binparse::Len {
        binparse::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn a_start_offset(&mut self) -> binparse::Len {
        binparse::Len::ZERO
    }
    pub fn a_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.a_start_offset().bits()..self.a_end_offset().bits()
    }
    #[allow(dead_code)]
    fn a_fatal_check(&mut self) -> Result<(), ::binparse::ParseError> {
        {
            let len = self.a_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::binparse::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn a_recoverable_check(&mut self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn b_present_node(&mut self) -> ::binparse::FieldNode<'a> {
        let bit_range = self.b_bit_range();
        ::binparse::FieldNode::new(
            "b",
            "u16",
            bit_range.clone(),
            ::binparse::Value::UInt(u128::from(self.b())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn b(&mut self) -> u16 {
        u16::from_be_bytes(self.data[1usize..3usize].try_into().unwrap())
    }
    pub fn b_end_offset(&mut self) -> binparse::Len {
        binparse::Len {
            byte: 3usize,
            bit: 0usize,
        }
    }
    pub fn b_start_offset(&mut self) -> binparse::Len {
        self.a_end_offset()
    }
    pub fn b_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.b_start_offset().bits()..self.b_end_offset().bits()
    }
    #[allow(dead_code)]
    fn b_fatal_check(&mut self) -> Result<(), ::binparse::ParseError> {
        {
            let len = self.b_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::binparse::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn b_recoverable_check(&mut self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
}

impl<'a> PairWriter<'a> {
    pub const SIZE: usize = 3usize;
    pub fn new(data: &'a mut [u8]) -> ::binparse::WriteResult<Self> {
        if data.len() < Self::SIZE {
            return Err(::binparse::WriteError::NotEnoughSpace {
                expected: Self::SIZE,
                got: data.len(),
            });
        }
        Ok(Self { data })
    }
    pub fn writer_over(data: &'a mut [u8]) -> ::binparse::WriteResult<Self> {
        if data.len() < Self::SIZE {
            return Err(::binparse::WriteError::NotEnoughSpace {
                expected: Self::SIZE,
                got: data.len(),
            });
        }
        Ok(Self { data })
    }
    pub fn set_a(&mut self, value: u8) -> &mut Self {
        self.data[0usize] = value;
        self
    }
    pub fn set_b(&mut self, value: u16) -> &mut Self {
        self.data[1usize..3usize].copy_from_slice(&value.to_be_bytes());
        self
    }
    pub fn write_into(
        data: &'a mut [u8],
        content: &PairContent,
    ) -> ::binparse::WriteResult<usize> {
        let mut w = Self::new(data)?;
        w.set_a(content.a);
        w.set_b(content.b);
        Ok(Self::SIZE)
    }
    pub fn to_vec(content: &PairContent) -> ::std::vec::Vec<u8> {
        let mut buf = ::std::vec![0u8; Self::SIZE];
        let _ = PairWriter::write_into(&mut buf, content);
        buf
    }
}

impl<'a> Rec<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::binparse::ParseError> {
        let mut me = Self { data };
        me.n_fatal_check()?;
        me.n_recoverable_check()?;
        me.items_fatal_check()?;
        me.items_recoverable_check()?;
        me.crc_fatal_check()?;
        me.crc_recoverable_check()?;
        let len = me.crc_end_offset();
        if len.bit != 0 {
            return Err(::binparse::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::binparse::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::binparse::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.n_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.items_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.crc_present_node());
        }
        let mut root = ::binparse::FieldNode::new(
                "Rec",
                "Rec",
                0usize..self.crc_end_offset().bits(),

        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::binparse::FieldNode<'a> {

        let mut children: ::std::vec::Vec<::binparse::FieldNode<'a>> = ::std::vec::Vec::new();

            match me.n_fatal_check() {

                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.items_fatal_check() {
                Err(error) => {
                    let start = me.items_start_offset().bits();
                    children
                        .push(
                            ::binparse::FieldNode::new(
                                    "items",
                                    "[Pair]",

                                    ::binparse::Value::Opaque,
                                )
                                .with_status(::binparse::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.items_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.items_bit_range();

                                    ::binparse::FieldNode::new(
                                            "items",
                                            "[Pair]",
                                            bit_range,
                                            ::binparse::Value::Opaque,

        }

                    let start = me.crc_start_offset().bits();

                            ::binparse::FieldNode::new(

                                    "u16",

    pub fn handoff(&mut self) -> Option<::binparse::Handoff<'a>> {

    }
    #[allow(dead_code)]
    fn n_present_node(&mut self) -> ::binparse::FieldNode<'a> {

        ::binparse::FieldNode::new(
            "n",
            "u8",
            bit_range.clone(),
            ::binparse::Value::UInt(u128::from(self.n())),

    pub fn n(&mut self) -> u8 {
        self.data[0usize]

    pub fn n_end_offset(&mut self) -> binparse::Len {
        binparse::Len {
            byte: 1usize,
            bit: 0usize,

    #[allow(dead_code)]
    fn n_fatal_check(&mut self) -> Result<(), ::binparse::ParseError> {
        {
            let len = self.n_end_offset();

                return Err(::binparse::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn n_recoverable_check(&mut self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn items_present_node(&mut self) -> ::binparse::FieldNode<'a> {
        let bit_range = self.items_bit_range();
        {
            let mut elem_nodes = ::std::vec::Vec::new();
            if let Ok(iter) = self.items() {
                let mut start = bit_range.start;
                for (i, elem) in iter.enumerate() {
                    match elem {
                        Ok(mut value) => {
                            let end = start.saturating_add(value.b_end_offset().bits());
                            elem_nodes
                                .push(
                                    value.field_tree().renamed(i.to_string()).shifted(start),
                                );
                            start = end;
                        }
                        Err(error) => {
                            elem_nodes
                                .push(
                                    ::binparse::FieldNode::new(
                                            i.to_string(),
                                            "Pair",
                                            start..start,
                                            ::binparse::Value::Opaque,
                                        )
                                        .with_status(::binparse::Status::Error(error)),
                                );
                        }
                    }
                }
            }
            ::binparse::FieldNode::new(
                    "items",
                    "[Pair]",
                    bit_range.clone(),
                    ::binparse::Value::Array,
                )
                .with_children(elem_nodes)
        }
    }
    #[allow(clippy::identity_op)]
    pub fn items(&mut self) -> ::binparse::ParseResult<Rec_items_Iterator<'a>> {
        Ok(Rec_items_Iterator {
            idx: 0,
            count: self.n() as usize,
            data: &self.data[1usize..],
        })
    }
    pub fn items_end_offset(&mut self) -> binparse::Len {
        ::binparse::Len {
            byte: 1usize,
            bit: 0usize,
        }
            + ({
                ::binparse::Len {
                    byte: 3usize,
                    bit: 0usize,
                } * (self.n() as usize)
            })
    }
    pub fn items_start_offset(&mut self) -> binparse::Len {
        self.n_end_offset()
    }
    pub fn items_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.items_start_offset().bits()..self.items_end_offset().bits()
    }
    #[allow(dead_code)]
    fn items_fatal_check(&mut self) -> Result<(), ::binparse::ParseError> {
        {
            let len = self.items_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::binparse::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn items_recoverable_check(&mut self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn crc_present_node(&mut self) -> ::binparse::FieldNode<'a> {
        let bit_range = self.crc_bit_range();
        ::binparse::FieldNode::new(
            "crc",
            "u16",
            bit_range.clone(),
            ::binparse::Value::UInt(u128::from(self.crc())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn crc(&mut self) -> u16 {
        {
            let offset = ::binparse::Len {
                byte: 1usize,
                bit: 0usize,
            }
                + ({
                    ::binparse::Len {
                        byte: 3usize,
                        bit: 0usize,
                    } * (self.n() as usize)
                });
            debug_assert!(offset.bit == 0, "primitive requires byte alignment");
            let start = offset.byte;
            let end = start + 2usize;
            u16::from_be_bytes(self.data[start..end].try_into().unwrap())
        }
    }
    pub fn crc_end_offset(&mut self) -> binparse::Len {
        ::binparse::Len {
            byte: 2usize,
            bit: 0usize,
        }
            + ({
                ::binparse::Len {
                    byte: 1usize,
                    bit: 0usize,
                }
                    + ({
                        ::binparse::Len {
                            byte: 3usize,
                            bit: 0usize,
                        } * (self.n() as usize)
                    })
            })
    }
    pub fn crc_start_offset(&mut self) -> binparse::Len {
        self.items_end_offset()
    }
    pub fn crc_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.crc_start_offset().bits()..self.crc_end_offset().bits()
    }
    #[allow(dead_code)]
    fn crc_fatal_check(&mut self) -> Result<(), ::binparse::ParseError> {
        {
            let len = self.crc_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::binparse::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn crc_recoverable_check(&mut self) -> Result<(), ::binparse::ParseError> {
        Ok(())
    }
}

impl<'a> RecWriter<'a> {
    pub fn encoded_len(lens: &RecLens) -> usize {
        1usize + (lens.items * 3usize) + 2usize
    }
    pub fn new(data: &'a mut [u8], lens: RecLens) -> ::binparse::WriteResult<Self> {
        let need = Self::encoded_len(&lens);
        if data.len() < need {
            return Err(::binparse::WriteError::NotEnoughSpace {
                expected: need,
                got: data.len(),
            });
        }
        if lens.items > (u8::MAX as usize) {
            return Err(::binparse::WriteError::ValueTooLarge {
                field: "Rec.n",
                value: lens.items,
                max: u8::MAX as usize,
            });
        }
        let mut me = Self { data, lens };
        me.write_len();
        Ok(me)
    }
    pub fn writer_over(data: &'a mut [u8]) -> ::binparse::WriteResult<Self> {
        let lens = {
            let (mut view, _) = Rec::parse(data)
                .map_err(|_| ::binparse::WriteError::InvalidContent)?;
            let region_bytes = view.items_end_offset().byte
                - view.items_start_offset().byte;
            RecLens {
                items: region_bytes / 3usize,
            }
        };
        let need = Self::encoded_len(&lens);
        if data.len() < need {
            return Err(::binparse::WriteError::NotEnoughSpace {
                expected: need,
                got: data.len(),
            });
        }
        Ok(Self { data, lens })
    }
    fn write_len(&mut self) {
        self.data[0usize] = self.lens.items as u8;
    }
    pub fn set_crc(&mut self, value: u16) -> &mut Self {
        let base = 1usize + (self.lens.items * 3usize);
        let trailer = &mut self.data[base..];
        trailer[0usize..2usize].copy_from_slice(&value.to_be_bytes());
        self
    }
    pub fn write_into(
        data: &'a mut [u8],
        content: &RecContent,
    ) -> ::binparse::WriteResult<usize> {
        let lens = RecLens {
            items: content.items.len(),
        };
        let need = Self::encoded_len(&lens);
        let mut w = Self::new(data, lens)?;
        let base = 1usize;
        for (i, c) in content.items.iter().enumerate() {
            let start = base + i * 3usize;
            PairWriter::write_into(&mut w.data[start..start + 3usize], c)?;
        }
        let trailer_base = 1usize + (w.lens.items * 3usize);
        {
            let value: u16 = content.crc;
            let trailer = &mut w.data[trailer_base..];
            trailer[0usize..2usize].copy_from_slice(&value.to_be_bytes());
        }
        Ok(need)
    }
    pub fn to_vec(content: &RecContent) -> ::std::vec::Vec<u8> {
        let lens = RecLens {
            items: content.items.len(),
        };
        let mut buf = ::std::vec![0u8; Self::encoded_len(& lens)];
        let _ = RecWriter::write_into(&mut buf, content);
        buf
    }
}

pub struct Pair<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}

pub struct PairContent {
    pub a: u8,
    pub b: u16,
}

pub struct PairWriter<'a> {
    data: &'a mut [u8],
}

pub struct Rec<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}

pub struct RecContent<'a> {
    pub items: &'a [PairContent],
    pub crc: u16,
}

pub struct RecWriter<'a> {
    data: &'a mut [u8],
    lens: RecLens,
}
