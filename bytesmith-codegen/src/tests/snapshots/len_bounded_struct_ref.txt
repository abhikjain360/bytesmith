impl<'a> ::bytesmith::Dissect<'a> for Inner<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        Inner::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        Inner::handoff(self)
    }
}

impl<'a> ::bytesmith::Dissect<'a> for Tlv<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        Tlv::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        Tlv::handoff(self)
    }
}

impl<'a> Inner<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.a_fatal_check()?;
        me.a_recoverable_check()?;
        me.b_fatal_check()?;
        me.b_recoverable_check()?;
        let len = me.b_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.a_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.b_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "Inner",
                "Inner",
                0usize..self.b_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.a_fatal_check() {
                Err(error) => {
                    let start = me.a_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "a",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.a_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.a_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "a",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.a_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.b_fatal_check() {
                Err(error) => {
                    let start = me.b_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "b",
                                    "u16",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.b_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.b_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "b",
                                            "u16",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.b_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "Inner",
                "Inner",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn a_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.a_bit_range();
        ::bytesmith::FieldNode::new(
            "a",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.a())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn a(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn a_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn a_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn a_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.a_start_offset().bits()..self.a_end_offset().bits()
    }
    #[allow(dead_code)]
    fn a_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.a_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn a_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn b_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.b_bit_range();
        ::bytesmith::FieldNode::new(
            "b",
            "u16",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.b())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn b(&mut self) -> u16 {
        u16::from_be_bytes(self.data[1usize..3usize].try_into().unwrap())
    }
    pub fn b_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 3usize,
            bit: 0usize,
        }
    }
    pub fn b_start_offset(&mut self) -> bytesmith::Len {
        self.a_end_offset()
    }
    pub fn b_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.b_start_offset().bits()..self.b_end_offset().bits()
    }
    #[allow(dead_code)]
    fn b_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.b_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn b_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}

impl<'a> Tlv<'a> {

        let mut me = Self { data };
        me.tag_fatal_check()?;
        me.tag_recoverable_check()?;
        me.len_fatal_check()?;
        me.len_recoverable_check()?;
        me.value_fatal_check()?;
        me.value_recoverable_check()?;
        me.after_fatal_check()?;
        me.after_recoverable_check()?;
        let len = me.after_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.tag_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.len_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.value_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.after_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "Tlv",
                "Tlv",
                0usize..self.after_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }

                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.after_fatal_check() {

                    let start = me.after_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "after",
                                    "u8",

        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(

    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn tag_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.tag_bit_range();
        ::bytesmith::FieldNode::new(
            "tag",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.tag())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn tag(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn tag_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn tag_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn tag_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.tag_start_offset().bits()..self.tag_end_offset().bits()
    }
    #[allow(dead_code)]
    fn tag_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.tag_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn tag_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {

    }
    #[allow(dead_code)]
    fn len_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.len_bit_range();
        ::bytesmith::FieldNode::new(
            "len",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.len())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn len(&mut self) -> u8 {
        self.data[1usize]
    }
    pub fn len_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 2usize,
            bit: 0usize,
        }
    }
    pub fn len_start_offset(&mut self) -> bytesmith::Len {
        self.tag_end_offset()
    }
    pub fn len_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.len_start_offset().bits()..self.len_end_offset().bits()
    }
    #[allow(dead_code)]
    fn len_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.len_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn len_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn value_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.value_bit_range();
        match self.value() {
            Ok(mut value) => {
                let inner = value.field_tree().renamed("value").shifted(bit_range.start);
                let consumed = inner.bit_range.end.min(bit_range.end);
                let mut node = inner.with_bit_range(bit_range.clone());
                if let Ok(rest) = self.value_rest() && !rest.is_empty() {
                    node.children
                        .push(
                            ::bytesmith::FieldNode::new(
                                "rest",
                                "[u8]",
                                consumed..bit_range.end,
                                ::bytesmith::Value::Bytes(rest),
                            ),
                        );
                }
                node
            }
            Err(error) => {
                ::bytesmith::FieldNode::new(
                        "value",
                        "Inner",
                        bit_range.clone(),
                        ::bytesmith::Value::Opaque,
                    )
                    .with_status(::bytesmith::Status::Error(error))
            }
        }
    }
    pub fn value_rest(&mut self) -> ::bytesmith::ParseResult<&'a [u8]> {
        let start = 2usize;
        let end = start.saturating_add(self.len() as usize);
        Inner::parse(&self.data[start..end]).map(|(_, rest)| rest)
    }
    #[allow(clippy::identity_op)]
    pub fn value(&mut self) -> ::bytesmith::ParseResult<Inner<'a>> {
        let start = 2usize;
        let end = start.saturating_add(self.len() as usize);
        Inner::parse(&self.data[start..end]).map(|(value, _)| value)
    }
    pub fn value_end_offset(&mut self) -> bytesmith::Len {
        ::bytesmith::Len {
            byte: 2usize,
            bit: 0usize,
        }
            + ({
                ::bytesmith::Len {
                    byte: self.len() as usize,
                    bit: 0,
                }
            })
    }
    pub fn value_start_offset(&mut self) -> bytesmith::Len {
        self.len_end_offset()
    }
    pub fn value_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.value_start_offset().bits()..self.value_end_offset().bits()
    }
    #[allow(dead_code)]
    fn value_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.value_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn value_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn after_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.after_bit_range();
        ::bytesmith::FieldNode::new(
            "after",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.after())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn after(&mut self) -> u8 {
        {
            let offset = ::bytesmith::Len {
                byte: 2usize,
                bit: 0usize,
            }
                + ({
                    ::bytesmith::Len {
                        byte: self.len() as usize,
                        bit: 0,
                    }
                });
            debug_assert!(offset.bit == 0, "primitive requires byte alignment");
            self.data[offset.byte]
        }
    }
    pub fn after_end_offset(&mut self) -> bytesmith::Len {
        ::bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
            + ({
                ::bytesmith::Len {
                    byte: 2usize,
                    bit: 0usize,
                }
                    + ({
                        ::bytesmith::Len {
                            byte: self.len() as usize,
                            bit: 0,
                        }
                    })
            })
    }
    pub fn after_start_offset(&mut self) -> bytesmith::Len {
        self.value_end_offset()
    }
    pub fn after_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.after_start_offset().bits()..self.after_end_offset().bits()
    }
    #[allow(dead_code)]
    fn after_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.after_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn after_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}
