#[allow(non_camel_case_types)]
pub struct VarFrame_body_Iterator<'a> {
    idx: usize,
    count: usize,
    data: &'a [u8],
}

#[derive(Clone, Copy)]
pub struct VarFrameLens {
    pub body: usize,
}

impl<'a> ::bytesmith::Dissect<'a> for VarFrame<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        VarFrame::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        VarFrame::handoff(self)
    }
}

impl<'a> ::std::iter::Iterator for VarFrame_body_Iterator<'a> {
    type Item = ::bytesmith::ParseResult<u8>;
    fn next(&mut self) -> std::option::Option<Self::Item> {
        if self.idx == self.count {
            return None;
        }
        self.idx += 1;
        let value = self.data[0];
        self.data = &self.data[1..];
        Some(Ok(value))
    }
}

impl<'a> VarFrame<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.tag_fatal_check()?;
        me.tag_recoverable_check()?;
        me.len_fatal_check()?;
        me.len_recoverable_check()?;
        me.body_fatal_check()?;
        me.body_recoverable_check()?;
        let len = me.body_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.tag_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.len_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.body_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "VarFrame",
                "VarFrame",
                0usize..self.body_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.tag_fatal_check() {
                Err(error) => {
                    let start = me.tag_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "tag",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.tag_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.tag_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "tag",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.tag_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.len_fatal_check() {
                Err(error) => {
                    let start = me.len_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "len",
                                    "[u8]",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.len_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.len_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "len",
                                            "[u8]",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.len_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.body_fatal_check() {
                Err(error) => {
                    let start = me.body_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "body",
                                    "[u8]",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.body_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.body_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "body",
                                            "[u8]",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.body_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "VarFrame",
                "VarFrame",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn tag_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.tag_bit_range();
        ::bytesmith::FieldNode::new(
            "tag",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.tag())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn tag(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn tag_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn tag_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn tag_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.tag_start_offset().bits()..self.tag_end_offset().bits()
    }
    #[allow(dead_code)]
    fn tag_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.tag_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn tag_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn len_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.len_bit_range();
        match self.len_raw() {
            Ok((value, _)) => {
                ::bytesmith::FieldNode::new(
                    "len",
                    "u64",
                    bit_range.clone(),
                    ::bytesmith::Value::UInt(u128::from(value)),
                )
            }
            Err(error) => {
                ::bytesmith::FieldNode::new(
                        "len",
                        "u64",
                        bit_range.clone(),
                        ::bytesmith::Value::Opaque,
                    )
                    .with_status(::bytesmith::Status::Error(error))
            }
        }
    }
    fn len_raw(&mut self) -> ::bytesmith::ParseResult<(u64, usize)> {
        let start = 1usize;
        let (value, consumed) = bytesmith::hooks::leb128_unsigned(
            &self.data[start..],
            ::bytesmith::HookContext {
                field: "VarFrame.len",
                offset: start,
                enclosing: self.data,
            },
        )?;
        let end = start.saturating_add(consumed);
        if end > self.data.len() {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: end,
                got: self.data.len(),
            });
        }
        Ok((value, consumed))
    }
    pub fn len(&mut self) -> ::bytesmith::ParseResult<u64> {
        self.len_raw().map(|(value, _)| value)
    }
    pub fn len_end_offset(&mut self) -> bytesmith::Len {
        ::bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
            + ({
                match self.len_raw() {
                    Ok((_, consumed)) => {
                        bytesmith::Len {
                            byte: consumed,
                            bit: 0,
                        }
                    }
                    Err(_) => bytesmith::Len::ZERO,
                }
            })
    }
    pub fn len_start_offset(&mut self) -> bytesmith::Len {
        self.tag_end_offset()
    }
    pub fn len_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.len_start_offset().bits()..self.len_end_offset().bits()
    }
    #[allow(dead_code)]
    fn len_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        self.len_raw()?;
        {
            let len = self.len_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn len_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn body_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.body_bit_range();
        {
            let mut elem_nodes = ::std::vec::Vec::new();
            if let Ok(iter) = self.body() {
                let mut start = bit_range.start;
                for (i, elem) in iter.enumerate() {
                    let end = start.saturating_add(8usize);
                    match elem {
                        Ok(value) => {
                            elem_nodes
                                .push(
                                    ::bytesmith::FieldNode::new(
                                        i.to_string(),
                                        "u8",
                                        start..end,
                                        ::bytesmith::Value::UInt(u128::from(value)),
                                    ),
                                )
                        }
                        Err(error) => {
                            elem_nodes
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            i.to_string(),
                                            "u8",
                                            start..start,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                    }
                    start = end;
                }
            }
            ::bytesmith::FieldNode::new(
                    "body",
                    "[u8]",
                    bit_range.clone(),
                    ::bytesmith::Value::Array,
                )
                .with_children(elem_nodes)
        }
    }
    #[allow(clippy::identity_op)]
    pub fn body(&mut self) -> ::bytesmith::ParseResult<VarFrame_body_Iterator<'a>> {
        Ok(VarFrame_body_Iterator {
            idx: 0,
            count: self.len().map(|v| v as usize).unwrap_or(0),
            data: &self
                .data[{
                let len = ::bytesmith::Len {
                    byte: 1usize,
                    bit: 0usize,
                }
                    + ({
                        match self.len_raw() {
                            Ok((_, consumed)) => {
                                bytesmith::Len {
                                    byte: consumed,
                                    bit: 0,
                                }
                            }
                            Err(_) => bytesmith::Len::ZERO,
                        }
                    });
                if len.bit > 0 {
                    return Err(::bytesmith::ParseError::UnalignedLength(len));
                }
                len.byte
            }..],
        })
    }
    pub fn body_end_offset(&mut self) -> bytesmith::Len {
        ({
            ::bytesmith::Len {
                byte: 1usize,
                bit: 0usize,
            }
                + ({
                    match self.len_raw() {
                        Ok((_, consumed)) => {
                            bytesmith::Len {
                                byte: consumed,
                                bit: 0,
                            }
                        }
                        Err(_) => bytesmith::Len::ZERO,
                    }
                })
        })
            + ({
                ::bytesmith::Len {
                    byte: 1usize,
                    bit: 0usize,
                } * (self.len().map(|v| v as usize).unwrap_or(0))
            })
    }
    pub fn body_start_offset(&mut self) -> bytesmith::Len {
        self.len_end_offset()
    }
    pub fn body_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.body_start_offset().bits()..self.body_end_offset().bits()
    }
    #[allow(dead_code)]
    fn body_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.body_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn body_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}

impl<'a> VarFrameWriter<'a> {
    pub fn encoded_len(lens: &VarFrameLens) -> usize {
        1usize + bytesmith::hooks::leb128_unsigned_len(lens.body as u64) + lens.body
    }
    pub fn new(
        data: &'a mut [u8],
        lens: VarFrameLens,
    ) -> ::bytesmith::WriteResult<Self> {
        let need = Self::encoded_len(&lens);
        if data.len() < need {
            return Err(::bytesmith::WriteError::NotEnoughSpace {
                expected: need,
                got: data.len(),
            });
        }
        let len_width = bytesmith::hooks::write_leb128_unsigned(
            lens.body as u64,
            &mut data[1usize..],
        )?;
        Ok(Self { data, lens, len_width })
    }
    pub fn set_tag(&mut self, value: u8) -> &mut Self {
        self.data[0usize] = value;
        self
    }
    pub fn body_mut(&mut self) -> &mut [u8] {
        let off = 1usize + self.len_width;
        &mut self.data[off..off + self.lens.body]
    }
    pub fn write_into(
        data: &'a mut [u8],
        content: &VarFrameContent,
    ) -> ::bytesmith::WriteResult<usize> {
        let lens = VarFrameLens {
            body: content.body.len(),
        };
        let need = Self::encoded_len(&lens);
        let mut w = Self::new(data, lens)?;
        w.set_tag(content.tag);
        w.body_mut().copy_from_slice(content.body);
        Ok(need)
    }
    pub fn to_vec(content: &VarFrameContent) -> ::std::vec::Vec<u8> {
        let lens = VarFrameLens {
            body: content.body.len(),
        };
        let mut buf = ::std::vec![0u8; Self::encoded_len(& lens)];
        let _ = VarFrameWriter::write_into(&mut buf, content);
        buf
    }
}

pub struct VarFrame<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}

pub struct VarFrameContent<'a> {
    pub tag: u8,
    pub body: &'a [u8],
}

pub struct VarFrameWriter<'a> {
    data: &'a mut [u8],
    lens: VarFrameLens,
    len_width: usize,
}
