    idx: usize,
    count: usize,
    data: &'a [u8],
    bit_offset: usize,
}

        BitArray::field_tree(self)

        BitArray::handoff(self)
    }

impl<'a> ::std::iter::Iterator for BitArray_nibbles_Iterator<'a> {
    type Item = ::bytesmith::ParseResult<u8>;
    fn next(&mut self) -> std::option::Option<Self::Item> {
        if self.idx == self.count {
            return None;
        }
        self.idx += 1;
        let byte_idx = self.bit_offset / 8;
        let bit_idx = self.bit_offset % 8;
        let value = if bit_idx + 4usize <= 8 {
            let mask = (1u8 << 4usize) - 1;
            (self.data[byte_idx] >> (8 - bit_idx - 4usize)) & mask
        } else {
            let bits_in_first = 8 - bit_idx;
            let bits_in_second = 4usize - bits_in_first;
            let first_mask = (1u8 << bits_in_first) - 1;
            let first_part = self.data[byte_idx] & first_mask;
            let second_part = self.data[byte_idx + 1] >> (8 - bits_in_second);
            (first_part << bits_in_second) | second_part
        };
        self.bit_offset += 4usize;
        Some(Ok(value))
    }
}

impl<'a> BitArray<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.nibbles_fatal_check()?;
        me.nibbles_recoverable_check()?;
        let len = me.nibbles_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.nibbles_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "BitArray",
                "BitArray",
                0usize..self.nibbles_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.nibbles_fatal_check() {
                Err(error) => {
                    let start = me.nibbles_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "nibbles",
                                    "[b<4>]",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.nibbles_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.nibbles_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "nibbles",
                                            "[b<4>]",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.nibbles_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "BitArray",
                "BitArray",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn nibbles_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.nibbles_bit_range();
        {
            let mut elem_nodes = ::std::vec::Vec::new();
            if let Ok(iter) = self.nibbles() {
                let mut start = bit_range.start;
                for (i, elem) in iter.enumerate() {
                    let end = start.saturating_add(4usize);
                    match elem {
                        Ok(value) => {
                            elem_nodes
                                .push(
                                    ::bytesmith::FieldNode::new(
                                        i.to_string(),
                                        "b<4>",
                                        start..end,
                                        ::bytesmith::Value::UInt(u128::from(value)),
                                    ),
                                )
                        }
                        Err(error) => {
                            elem_nodes
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            i.to_string(),
                                            "b<4>",
                                            start..start,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                    }
                    start = end;
                }
            }
            ::bytesmith::FieldNode::new(
                    "nibbles",
                    "[b<4>]",
                    bit_range.clone(),
                    ::bytesmith::Value::Array,
                )
                .with_children(elem_nodes)
        }
    }
    #[allow(clippy::identity_op)]
    pub fn nibbles(

        Ok(BitArray_nibbles_Iterator {
            idx: 0,
            count: 4usize,

            bit_offset: 0,
        })
    }
    pub fn nibbles_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 2usize,
            bit: 0usize,
        }
    }
    pub fn nibbles_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn nibbles_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.nibbles_start_offset().bits()..self.nibbles_end_offset().bits()
    }
    #[allow(dead_code)]
    fn nibbles_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.nibbles_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn nibbles_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}
