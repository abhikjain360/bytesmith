impl<'a> ::bytesmith::Dissect<'a> for LsbBits<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        LsbBits::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        LsbBits::handoff(self)
    }
}

impl<'a> LsbBits<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.low_fatal_check()?;
        me.low_recoverable_check()?;
        me.high_fatal_check()?;
        me.high_recoverable_check()?;
        let len = me.high_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.low_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.high_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "LsbBits",
                "LsbBits",
                0usize..self.high_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.low_fatal_check() {
                Err(error) => {
                    let start = me.low_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "low",
                                    "b<3>",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.low_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.low_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "low",
                                            "b<3>",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.low_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.high_fatal_check() {
                Err(error) => {
                    let start = me.high_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "high",
                                    "b<5>",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.high_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.high_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "high",
                                            "b<5>",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.high_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "LsbBits",
                "LsbBits",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn low_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.low_bit_range();
        ::bytesmith::FieldNode::new(
            "low",
            "b<3>",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.low())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn low(&mut self) -> u8 {
        (self.data[0usize] >> 0usize) & 7u8
    }
    pub fn low_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 0usize,
            bit: 3usize,
        }
    }
    pub fn low_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn low_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.low_start_offset().bits()..self.low_end_offset().bits()
    }
    #[allow(dead_code)]
    fn low_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.low_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn low_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn high_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.high_bit_range();
        ::bytesmith::FieldNode::new(
            "high",
            "b<5>",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.high())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn high(&mut self) -> u8 {
        (self.data[0usize] >> 3usize) & 31u8
    }
    pub fn high_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn high_start_offset(&mut self) -> bytesmith::Len {
        self.low_end_offset()
    }
    pub fn high_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.high_start_offset().bits()..self.high_end_offset().bits()
    }
    #[allow(dead_code)]
    fn high_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.high_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn high_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}

pub struct LsbBits<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}
