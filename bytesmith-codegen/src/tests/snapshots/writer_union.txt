#[allow(non_camel_case_types)]
pub enum Packet_body<'a> {
    Connect(Packet_body_Connect<'a>),
    Connack(Packet_body_Connack<'a>),
    Unknown(Packet_body_Unknown<'a>),
}

#[allow(non_camel_case_types)]
pub struct Packet_body_Connack<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}

#[allow(non_camel_case_types)]
pub struct Packet_body_Connect<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}

#[allow(non_camel_case_types)]
pub struct Packet_body_Unknown<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}

impl<'a> ::bytesmith::Dissect<'a> for Packet<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        Packet::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        Packet::handoff(self)
    }
}

impl<'a> ConnackWriter<'a> {
    pub const SIZE: usize = 2usize;
    pub fn new(data: &'a mut [u8]) -> ::bytesmith::WriteResult<Self> {
        if data.len() < Self::SIZE {
            return Err(::bytesmith::WriteError::NotEnoughSpace {
                expected: Self::SIZE,
                got: data.len(),
            });
        }
        Ok(Self { data })
    }
    pub fn writer_over(data: &'a mut [u8]) -> ::bytesmith::WriteResult<Self> {
        if data.len() < Self::SIZE {
            return Err(::bytesmith::WriteError::NotEnoughSpace {
                expected: Self::SIZE,
                got: data.len(),
            });
        }
        Ok(Self { data })
    }
    pub fn set_ack(&mut self, value: u8) -> &mut Self {
        self.data[0usize] = value;
        self
    }
    pub fn set_code(&mut self, value: u8) -> &mut Self {
        self.data[1usize] = value;
        self
    }
    pub fn write_into(
        data: &'a mut [u8],
        content: &ConnackContent,
    ) -> ::bytesmith::WriteResult<usize> {
        let mut w = Self::new(data)?;
        w.set_ack(content.ack);
        w.set_code(content.code);
        Ok(Self::SIZE)
    }
    pub fn to_vec(content: &ConnackContent) -> ::std::vec::Vec<u8> {
        let mut buf = ::std::vec![0u8; Self::SIZE];
        let _ = ConnackWriter::write_into(&mut buf, content);
        buf
    }
}

impl<'a> ConnectWriter<'a> {
    pub const SIZE: usize = 2usize;
    pub fn new(data: &'a mut [u8]) -> ::bytesmith::WriteResult<Self> {

            return Err(::bytesmith::WriteError::NotEnoughSpace {
                expected: Self::SIZE,
                got: data.len(),
            });
        }
        Ok(Self { data })

    pub fn set_keep_alive(&mut self, value: u16) -> &mut Self {

impl<'a> Packet<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.kind_fatal_check()?;
        me.kind_recoverable_check()?;
        me.body_fatal_check()?;
        me.body_recoverable_check()?;
        let len = me.body_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.kind_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.body_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "Packet",
                "Packet",
                0usize..self.body_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.kind_fatal_check() {
                Err(error) => {
                    let start = me.kind_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "kind",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.kind_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.kind_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "kind",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.kind_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.body_fatal_check() {
                Err(error) => {
                    let start = me.body_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "body",
                                    "union",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.body_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.body_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "body",
                                            "union",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.body_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "Packet",
                "Packet",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn kind_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.kind_bit_range();
        ::bytesmith::FieldNode::new(
            "kind",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.kind())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn kind(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn kind_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn kind_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn kind_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.kind_start_offset().bits()..self.kind_end_offset().bits()
    }
    #[allow(dead_code)]
    fn kind_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.kind_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn kind_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn body_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.body_bit_range();
        match self.body() {
            Ok(Packet_body::Connect(mut value)) => {
                let inner = value
                    .field_tree()
                    .renamed("Connect")
                    .shifted(bit_range.start);
                ::bytesmith::FieldNode::new(
                        "body",
                        "union",
                        bit_range.clone(),
                        ::bytesmith::Value::UnionVariant("Connect"),
                    )
                    .with_children(::std::vec![inner])
            }
            Ok(Packet_body::Connack(mut value)) => {
                let inner = value
                    .field_tree()
                    .renamed("Connack")
                    .shifted(bit_range.start);
                ::bytesmith::FieldNode::new(
                        "body",
                        "union",
                        bit_range.clone(),
                        ::bytesmith::Value::UnionVariant("Connack"),
                    )
                    .with_children(::std::vec![inner])
            }
            Ok(Packet_body::Unknown(mut value)) => {
                let inner = value
                    .field_tree()
                    .renamed("Unknown")
                    .shifted(bit_range.start);
                ::bytesmith::FieldNode::new(
                        "body",
                        "union",
                        bit_range.clone(),
                        ::bytesmith::Value::UnionVariant("Unknown"),
                    )
                    .with_children(::std::vec![inner])
            }
            Err(error) => {
                ::bytesmith::FieldNode::new(
                        "body",
                        "union",
                        bit_range.clone(),
                        ::bytesmith::Value::Opaque,
                    )
                    .with_status(::bytesmith::Status::Error(error))
            }
        }
    }
    fn body_union_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        match self.kind() as usize {
            1 => {
                Packet_body_Connect::parse(&self.data[(1usize).min(self.data.len())..])?;
            }
            2 => {
                Packet_body_Connack::parse(&self.data[(1usize).min(self.data.len())..])?;
            }
            _ => {
                Packet_body_Unknown::parse(&self.data[(1usize).min(self.data.len())..])?;
            }
        }
        Ok(())
    }
    #[allow(clippy::identity_op)]
    pub fn body(&mut self) -> ::bytesmith::ParseResult<Packet_body<'a>> {
        match self.kind() as usize {
            1 => {
                Packet_body_Connect::parse(&self.data[1usize..])
                    .map(|(value, _)| Packet_body::Connect(value))
            }
            2 => {
                Packet_body_Connack::parse(&self.data[1usize..])
                    .map(|(value, _)| Packet_body::Connack(value))
            }
            _ => {
                Packet_body_Unknown::parse(&self.data[1usize..])
                    .map(|(value, _)| Packet_body::Unknown(value))
            }
        }
    }
    pub fn body_end_offset(&mut self) -> bytesmith::Len {
        ::bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
            + ({
                match self.kind() as usize {
                    1 => {
                        ::bytesmith::Len {
                            byte: 2usize,
                            bit: 0usize,
                        }
                    }
                    2 => {
                        ::bytesmith::Len {
                            byte: 2usize,
                            bit: 0usize,
                        }
                    }
                    _ => {
                        ::bytesmith::Len {
                            byte: 0usize,
                            bit: 0usize,
                        }
                    }
                }
            })
    }
    pub fn body_start_offset(&mut self) -> bytesmith::Len {
        self.kind_end_offset()
    }
    pub fn body_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.body_start_offset().bits()..self.body_end_offset().bits()
    }
    #[allow(dead_code)]
    fn body_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        self.body_union_check()?;
        {
            let len = self.body_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn body_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}

impl<'a> PacketWriter<'a> {
    pub fn encoded_len(content: &PacketContent) -> usize {
        1usize
            + match &content.body {
                PacketBodyContent::Connect(_) => ConnectWriter::SIZE,
                PacketBodyContent::Connack(_) => ConnackWriter::SIZE,
            }
    }
    pub fn write_into(
        data: &'a mut [u8],
        content: &PacketContent,
    ) -> ::bytesmith::WriteResult<usize> {
        let need = Self::encoded_len(content);
        if data.len() < need {
            return Err(::bytesmith::WriteError::NotEnoughSpace {
                expected: need,
                got: data.len(),
            });
        }
        let mut w = Self { data };
        match &content.body {
            PacketBodyContent::Connect(c) => {
                let value: u8 = 1;
                w.data[0usize] = value;
                let end = 1usize + ConnectWriter::SIZE;
                ConnectWriter::write_into(&mut w.data[1usize..end], c)?;
            }
            PacketBodyContent::Connack(c) => {
                let value: u8 = 2;
                w.data[0usize] = value;
                let end = 1usize + ConnackWriter::SIZE;
                ConnackWriter::write_into(&mut w.data[1usize..end], c)?;
            }
        }
        Ok(need)
    }
    pub fn to_vec(content: &PacketContent) -> ::std::vec::Vec<u8> {
        let mut buf = ::std::vec![0u8; Self::encoded_len(content)];
        let _ = PacketWriter::write_into(&mut buf, content);
        buf
    }
}

    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.keep_alive_fatal_check() {
                Err(error) => {
                    let start = me.keep_alive_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "keep_alive",
                                    "u16",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.keep_alive_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.keep_alive_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "keep_alive",
                                            "u16",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.keep_alive_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "Packet_body_Connect",
                "Packet_body_Connect",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {

        }

    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn keep_alive_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.keep_alive_bit_range();
        ::bytesmith::FieldNode::new(
            "keep_alive",
            "u16",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.keep_alive())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn keep_alive(&mut self) -> u16 {
        u16::from_be_bytes(self.data[0usize..2usize].try_into().unwrap())
    }
    pub fn keep_alive_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 2usize,
            bit: 0usize,
        }
    }
    pub fn keep_alive_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn keep_alive_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.keep_alive_start_offset().bits()..self.keep_alive_end_offset().bits()
    }
    #[allow(dead_code)]
    fn keep_alive_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.keep_alive_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn keep_alive_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
