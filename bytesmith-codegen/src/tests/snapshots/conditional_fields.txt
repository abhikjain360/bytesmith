impl<'a> ::bytesmith::Dissect<'a> for Cond<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        Cond::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        Cond::handoff(self)
    }
}

impl<'a> Cond<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.n_fatal_check()?;
        me.n_recoverable_check()?;
        if me.conditional_0_present() {
            me.x_fatal_check()?;
            me.x_recoverable_check()?;
        }
        if me.conditional_0_absent() {
            me.y_fatal_check()?;
            me.y_recoverable_check()?;
        }
        me.tail_fatal_check()?;
        me.tail_recoverable_check()?;
        let len = me.tail_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.n_present_node());
        }
        {
            let me = &mut *self;
            if me.conditional_0_present() {
                children.push(me.x_present_node());
            } else {
                children.push(me.x_absent_node());
            }
        }
        {
            let me = &mut *self;
            if me.conditional_0_absent() {
                children.push(me.y_present_node());
            } else {
                children.push(me.y_absent_node());
            }
        }
        {
            let me = &mut *self;
            children.push(me.tail_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "Cond",
                "Cond",
                0usize..self.tail_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.n_fatal_check() {
                Err(error) => {
                    let start = me.n_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "n",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.n_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.n_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "n",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.n_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            if me.conditional_0_present() {
                match me.x_fatal_check() {
                    Err(error) => {
                        let start = me.x_start_offset().bits();
                        children
                            .push(
                                ::bytesmith::FieldNode::new(
                                        "x",
                                        "u16",
                                        start..start,
                                        ::bytesmith::Value::Opaque,
                                    )
                                    .with_status(::bytesmith::Status::Error(error)),
                            );
                        fatal = Some(error);
                    }
                    Ok(()) => {
                        match me.x_recoverable_check() {
                            Err(error) => {
                                let bit_range = me.x_bit_range();
                                children
                                    .push(
                                        ::bytesmith::FieldNode::new(
                                                "x",
                                                "u16",
                                                bit_range,
                                                ::bytesmith::Value::Opaque,
                                            )
                                            .with_status(::bytesmith::Status::Error(error)),
                                    );
                            }
                            Ok(()) => {
                                children.push(me.x_present_node());
                            }
                        }
                    }
                }
            } else {
                children.push(me.x_absent_node());
            }
        }
        if fatal.is_none() {
            if me.conditional_0_absent() {
                match me.y_fatal_check() {
                    Err(error) => {
                        let start = me.y_start_offset().bits();
                        children
                            .push(
                                ::bytesmith::FieldNode::new(
                                        "y",
                                        "u8",
                                        start..start,
                                        ::bytesmith::Value::Opaque,
                                    )
                                    .with_status(::bytesmith::Status::Error(error)),
                            );
                        fatal = Some(error);
                    }
                    Ok(()) => {
                        match me.y_recoverable_check() {
                            Err(error) => {
                                let bit_range = me.y_bit_range();
                                children
                                    .push(
                                        ::bytesmith::FieldNode::new(
                                                "y",
                                                "u8",
                                                bit_range,
                                                ::bytesmith::Value::Opaque,
                                            )
                                            .with_status(::bytesmith::Status::Error(error)),
                                    );
                            }
                            Ok(()) => {
                                children.push(me.y_present_node());
                            }
                        }
                    }
                }
            } else {
                children.push(me.y_absent_node());
            }
        }
        if fatal.is_none() {
            match me.tail_fatal_check() {
                Err(error) => {
                    let start = me.tail_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "tail",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.tail_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.tail_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "tail",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.tail_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "Cond",
                "Cond",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn n_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.n_bit_range();
        ::bytesmith::FieldNode::new(
            "n",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.n())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn n(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn n_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn n_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn n_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.n_start_offset().bits()..self.n_end_offset().bits()
    }
    #[allow(dead_code)]
    fn n_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.n_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn n_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code, unused_parens)]
    fn conditional_0_present(&mut self) -> bool {
        ((self.n() as usize) == (1usize))
    }
    #[allow(dead_code)]
    fn x_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.x_bit_range();
        ::bytesmith::FieldNode::new(
            "x",
            "u16",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.x_raw())),
        )
    }
    #[allow(dead_code)]
    fn x_absent_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let start = self.x_start_offset().bits();
        ::bytesmith::FieldNode::new("x", "u16", start..start, ::bytesmith::Value::Absent)
            .hide()
    }
    #[allow(clippy::identity_op)]
    fn x_raw(&mut self) -> u16 {
        u16::from_be_bytes(self.data[1usize..3usize].try_into().unwrap())
    }
    pub fn x(&mut self) -> Option<u16> {
        if self.conditional_0_present() { Some(self.x_raw()) } else { None }
    }
    pub fn x_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 3usize,
            bit: 0usize,
        }
    }
    pub fn x_start_offset(&mut self) -> bytesmith::Len {
        self.n_end_offset()
    }
    pub fn x_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.x_start_offset().bits()..self.x_end_offset().bits()
    }
    #[allow(dead_code)]
    fn x_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.x_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn x_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code, unused_parens)]
    fn conditional_0_absent(&mut self) -> bool {
        !((self.n() as usize) == (1usize))
    }
    #[allow(dead_code)]
    fn y_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.y_bit_range();
        ::bytesmith::FieldNode::new(
            "y",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.y_raw())),
        )
    }
    #[allow(dead_code)]
    fn y_absent_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let start = self.y_start_offset().bits();
        ::bytesmith::FieldNode::new("y", "u8", start..start, ::bytesmith::Value::Absent)
            .hide()
    }
    #[allow(clippy::identity_op)]
    fn y_raw(&mut self) -> u8 {
        self.data[1usize]
    }
    pub fn y(&mut self) -> Option<u8> {
        if self.conditional_0_absent() { Some(self.y_raw()) } else { None }
    }
    pub fn y_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 2usize,
            bit: 0usize,
        }
    }
    pub fn y_start_offset(&mut self) -> bytesmith::Len {
        self.n_end_offset()
    }
    pub fn y_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.y_start_offset().bits()..self.y_end_offset().bits()
    }
    #[allow(dead_code)]
    fn y_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.y_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn y_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    fn conditional_0_end_offset(&mut self) -> bytesmith::Len {
        if self.conditional_0_present() {
            self.x_end_offset()
        } else {
            self.y_end_offset()
        }
    }
    #[allow(dead_code)]
    fn tail_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.tail_bit_range();
        ::bytesmith::FieldNode::new(
            "tail",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.tail())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn tail(&mut self) -> u8 {
        {
            let offset = self.conditional_0_end_offset();
            debug_assert!(offset.bit == 0, "primitive requires byte alignment");
            self.data[offset.byte]
        }
    }
    pub fn tail_end_offset(&mut self) -> bytesmith::Len {
        ::bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        } + ({ self.conditional_0_end_offset() })
    }
    pub fn tail_start_offset(&mut self) -> bytesmith::Len {
        self.conditional_0_end_offset()
    }
    pub fn tail_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.tail_start_offset().bits()..self.tail_end_offset().bits()
    }
    #[allow(dead_code)]
    fn tail_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.tail_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn tail_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}

pub struct Cond<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}
