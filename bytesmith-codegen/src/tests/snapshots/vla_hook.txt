impl<'a> ::bytesmith::Dissect<'a> for WithVlaHook<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        WithVlaHook::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        WithVlaHook::handoff(self)
    }
}

impl<'a> WithVlaHook<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.len_fatal_check()?;
        me.len_recoverable_check()?;
        me.name_fatal_check()?;
        me.name_recoverable_check()?;
        let len = me.name_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.len_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.name_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "WithVlaHook",
                "WithVlaHook",
                0usize..self.name_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.len_fatal_check() {
                Err(error) => {
                    let start = me.len_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "len",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.len_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.len_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "len",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.len_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.name_fatal_check() {
                Err(error) => {
                    let start = me.name_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "name",
                                    "[u8]",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.name_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.name_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "name",
                                            "[u8]",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.name_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "WithVlaHook",
                "WithVlaHook",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn len_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.len_bit_range();
        ::bytesmith::FieldNode::new(
            "len",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.len())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn len(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn len_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn len_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn len_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.len_start_offset().bits()..self.len_end_offset().bits()
    }
    #[allow(dead_code)]
    fn len_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.len_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn len_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn name_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.name_bit_range();
        match self.name_raw() {
            Ok((_, _)) => {
                ::bytesmith::FieldNode::new(
                    "name",
                    "String",
                    bit_range.clone(),
                    ::bytesmith::Value::Opaque,
                )
            }
            Err(error) => {
                ::bytesmith::FieldNode::new(
                        "name",
                        "String",
                        bit_range.clone(),
                        ::bytesmith::Value::Opaque,
                    )
                    .with_status(::bytesmith::Status::Error(error))
            }
        }
    }
    fn name_raw(&mut self) -> ::bytesmith::ParseResult<(String, usize)> {
        let start = 1usize;
        let (value, consumed) = parse_cstring(
            &self.data[start..],
            ::bytesmith::HookContext {
                field: "WithVlaHook.name",
                offset: start,
                enclosing: self.data,
            },
        )?;
        let end = start.saturating_add(consumed);
        if end > self.data.len() {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: end,
                got: self.data.len(),
            });
        }
        Ok((value, consumed))
    }
    pub fn name(&mut self) -> ::bytesmith::ParseResult<String> {
        self.name_raw().map(|(value, _)| value)
    }
    pub fn name_end_offset(&mut self) -> bytesmith::Len {
        ::bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
            + ({
                match self.name_raw() {
                    Ok((_, consumed)) => {
                        bytesmith::Len {
                            byte: consumed,
                            bit: 0,
                        }
                    }
                    Err(_) => bytesmith::Len::ZERO,
                }
            })
    }
    pub fn name_start_offset(&mut self) -> bytesmith::Len {
        self.len_end_offset()
    }
    pub fn name_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.name_start_offset().bits()..self.name_end_offset().bits()
    }
    #[allow(dead_code)]
    fn name_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        self.name_raw()?;
        {
            let len = self.name_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn name_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}

pub struct WithVlaHook<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}
