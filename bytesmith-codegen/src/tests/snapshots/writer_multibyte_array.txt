#[allow(non_camel_case_types)]
pub struct Samples_data_Iterator<'a> {
    idx: usize,
    count: usize,
    data: &'a [u8],
}

impl<'a> ::bytesmith::Dissect<'a> for Samples<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        Samples::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        Samples::handoff(self)
    }
}

impl<'a> ::std::iter::Iterator for Samples_data_Iterator<'a> {
    type Item = ::bytesmith::ParseResult<u16>;
    fn next(&mut self) -> std::option::Option<Self::Item> {
        if self.idx == self.count {
            return None;
        }
        self.idx += 1;
        let value = u16::from_be_bytes(self.data[..2usize].try_into().unwrap());
        self.data = &self.data[2usize..];
        Some(Ok(value))
    }
}

impl<'a> Samples<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.count_fatal_check()?;
        me.count_recoverable_check()?;
        me.data_fatal_check()?;
        me.data_recoverable_check()?;
        me.crc_fatal_check()?;
        me.crc_recoverable_check()?;
        let len = me.crc_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.count_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.data_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.crc_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "Samples",
                "Samples",
                0usize..self.crc_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.count_fatal_check() {
                Err(error) => {
                    let start = me.count_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "count",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.count_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.count_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "count",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.count_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.data_fatal_check() {
                Err(error) => {
                    let start = me.data_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "data",
                                    "[u16]",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.data_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.data_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "data",
                                            "[u16]",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.data_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.crc_fatal_check() {
                Err(error) => {
                    let start = me.crc_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "crc",
                                    "u16",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.crc_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.crc_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "crc",
                                            "u16",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.crc_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "Samples",
                "Samples",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn count_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.count_bit_range();
        ::bytesmith::FieldNode::new(
            "count",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.count())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn count(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn count_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn count_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn count_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.count_start_offset().bits()..self.count_end_offset().bits()
    }
    #[allow(dead_code)]
    fn count_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.count_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn count_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn data_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.data_bit_range();
        {
            let mut elem_nodes = ::std::vec::Vec::new();
            if let Ok(iter) = self.data() {
                let mut start = bit_range.start;
                for (i, elem) in iter.enumerate() {
                    let end = start.saturating_add(16usize);
                    match elem {
                        Ok(value) => {
                            elem_nodes
                                .push(
                                    ::bytesmith::FieldNode::new(
                                        i.to_string(),
                                        "u16",
                                        start..end,
                                        ::bytesmith::Value::UInt(u128::from(value)),
                                    ),
                                )
                        }
                        Err(error) => {
                            elem_nodes
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            i.to_string(),
                                            "u16",
                                            start..start,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                    }
                    start = end;
                }
            }
            ::bytesmith::FieldNode::new(
                    "data",
                    "[u16]",
                    bit_range.clone(),
                    ::bytesmith::Value::Array,
                )
                .with_children(elem_nodes)
        }
    }
    #[allow(clippy::identity_op)]
    pub fn data(&mut self) -> ::bytesmith::ParseResult<Samples_data_Iterator<'a>> {
        Ok(Samples_data_Iterator {
            idx: 0,
            count: 4usize,
            data: &self.data[1usize..],
        })
    }
    pub fn data_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 9usize,
            bit: 0usize,
        }
    }
    pub fn data_start_offset(&mut self) -> bytesmith::Len {
        self.count_end_offset()
    }
    pub fn data_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.data_start_offset().bits()..self.data_end_offset().bits()
    }
    #[allow(dead_code)]
    fn data_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.data_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn data_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn crc_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.crc_bit_range();
        ::bytesmith::FieldNode::new(
            "crc",
            "u16",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.crc())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn crc(&mut self) -> u16 {
        u16::from_be_bytes(self.data[9usize..11usize].try_into().unwrap())
    }
    pub fn crc_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 11usize,
            bit: 0usize,
        }
    }
    pub fn crc_start_offset(&mut self) -> bytesmith::Len {
        self.data_end_offset()
    }
    pub fn crc_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.crc_start_offset().bits()..self.crc_end_offset().bits()
    }
    #[allow(dead_code)]
    fn crc_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.crc_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn crc_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}

impl<'a> SamplesWriter<'a> {
    pub const SIZE: usize = 11usize;
    pub fn new(data: &'a mut [u8]) -> ::bytesmith::WriteResult<Self> {
        if data.len() < Self::SIZE {
            return Err(::bytesmith::WriteError::NotEnoughSpace {
                expected: Self::SIZE,
                got: data.len(),
            });
        }
        Ok(Self { data })
    }
    pub fn writer_over(data: &'a mut [u8]) -> ::bytesmith::WriteResult<Self> {
        if data.len() < Self::SIZE {
            return Err(::bytesmith::WriteError::NotEnoughSpace {
                expected: Self::SIZE,
                got: data.len(),
            });
        }
        Ok(Self { data })
    }
    pub fn set_count(&mut self, value: u8) -> &mut Self {
        self.data[0usize] = value;
        self
    }
    pub fn set_data(&mut self, value: [u16; 4usize]) -> &mut Self {
        for __i in 0..4usize {
            let __start = 1usize + __i * 2usize;
            self.data[__start..__start + 2usize]
                .copy_from_slice(&value[__i].to_be_bytes());
        }
        self
    }
    pub fn set_crc(&mut self, value: u16) -> &mut Self {
        self.data[9usize..11usize].copy_from_slice(&value.to_be_bytes());
        self
    }
    pub fn write_into(
        data: &'a mut [u8],
        content: &SamplesContent,
    ) -> ::bytesmith::WriteResult<usize> {
        let mut w = Self::new(data)?;
        w.set_count(content.count);
        w.set_data(content.data);
        w.set_crc(content.crc);
        Ok(Self::SIZE)
    }
    pub fn to_vec(content: &SamplesContent) -> ::std::vec::Vec<u8> {
        let mut buf = ::std::vec![0u8; Self::SIZE];
        let _ = SamplesWriter::write_into(&mut buf, content);
        buf
    }
}

pub struct Samples<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}

pub struct SamplesContent {
    pub count: u8,
    pub data: [u16; 4usize],
    pub crc: u16,
}

pub struct SamplesWriter<'a> {
    data: &'a mut [u8],
}
