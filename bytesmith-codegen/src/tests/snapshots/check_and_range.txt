impl<'a> ::bytesmith::Dissect<'a> for Checked<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        Checked::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        Checked::handoff(self)
    }
}

impl<'a> Checked<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.n_fatal_check()?;
        me.n_recoverable_check()?;
        me.m_fatal_check()?;
        me.m_recoverable_check()?;
        let len = me.m_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.n_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.m_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "Checked",
                "Checked",
                0usize..self.m_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.n_fatal_check() {
                Err(error) => {
                    let start = me.n_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "n",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.n_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.n_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "n",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.n_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.m_fatal_check() {
                Err(error) => {
                    let start = me.m_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "m",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.m_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.m_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "m",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.m_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "Checked",
                "Checked",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn n_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.n_bit_range();
        ::bytesmith::FieldNode::new(
            "n",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.n())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn n(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn n_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn n_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn n_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.n_start_offset().bits()..self.n_end_offset().bits()
    }
    #[allow(dead_code)]
    fn n_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.n_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn n_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn m_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.m_bit_range();
        ::bytesmith::FieldNode::new(
            "m",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.m())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn m(&mut self) -> u8 {
        self.data[1usize]
    }
    pub fn m_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 2usize,
            bit: 0usize,
        }
    }
    pub fn m_start_offset(&mut self) -> bytesmith::Len {
        self.n_end_offset()
    }
    pub fn m_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.m_start_offset().bits()..self.m_end_offset().bits()
    }
    #[allow(dead_code)]
    fn m_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.m_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(clippy::unnecessary_cast)]
    fn m_validate(&mut self) -> Result<(), ::bytesmith::ParseError> {
        if !((1usize)..=((self.n() as usize).saturating_add(1usize)))
            .contains(&(self.m() as usize))
        {
            return Err(::bytesmith::ParseError::ValidationFailed {
                field: "Checked.m",
                actual: self.m() as u128,
            });
        }
        if !((self.m() as usize) >= (self.n() as usize)) {
            return Err(::bytesmith::ParseError::ValidationFailed {
                field: "Checked.m",
                actual: self.m() as u128,
            });
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn m_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        self.m_validate()?;
        Ok(())
    }
}

pub struct Checked<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}
