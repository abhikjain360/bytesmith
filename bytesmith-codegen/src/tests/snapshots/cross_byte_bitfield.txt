impl<'a> ::bytesmith::Dissect<'a> for Cross<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        Cross::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        Cross::handoff(self)
    }
}

impl<'a> Cross<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.a_fatal_check()?;
        me.a_recoverable_check()?;
        me.b_fatal_check()?;
        me.b_recoverable_check()?;
        me.c_fatal_check()?;
        me.c_recoverable_check()?;
        let len = me.c_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.a_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.b_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.c_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "Cross",
                "Cross",
                0usize..self.c_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.a_fatal_check() {
                Err(error) => {
                    let start = me.a_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "a",
                                    "b<5>",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.a_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.a_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "a",
                                            "b<5>",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.a_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.b_fatal_check() {
                Err(error) => {
                    let start = me.b_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "b",
                                    "b<6>",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.b_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.b_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "b",
                                            "b<6>",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.b_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.c_fatal_check() {
                Err(error) => {
                    let start = me.c_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "c",
                                    "b<5>",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.c_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.c_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "c",
                                            "b<5>",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.c_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "Cross",
                "Cross",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn a_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.a_bit_range();
        ::bytesmith::FieldNode::new(
            "a",
            "b<5>",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.a())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn a(&mut self) -> u8 {
        (self.data[0usize] >> 3usize) & 31u8
    }
    pub fn a_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 0usize,
            bit: 5usize,
        }
    }
    pub fn a_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn a_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.a_start_offset().bits()..self.a_end_offset().bits()
    }
    #[allow(dead_code)]
    fn a_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.a_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn a_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn b_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.b_bit_range();
        ::bytesmith::FieldNode::new(
            "b",
            "b<6>",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.b())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn b(&mut self) -> u8 {
        {
            let first_part = self.data[0usize] & 7u8;
            let second_part = self.data[1usize] >> 5usize;
            (first_part << 3usize) | second_part
        }
    }
    pub fn b_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 3usize,
        }
    }
    pub fn b_start_offset(&mut self) -> bytesmith::Len {
        self.a_end_offset()
    }
    pub fn b_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.b_start_offset().bits()..self.b_end_offset().bits()
    }
    #[allow(dead_code)]
    fn b_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.b_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn b_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn c_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.c_bit_range();
        ::bytesmith::FieldNode::new(
            "c",
            "b<5>",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.c())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn c(&mut self) -> u8 {
        (self.data[1usize] >> 0usize) & 31u8
    }
    pub fn c_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 2usize,
            bit: 0usize,
        }
    }
    pub fn c_start_offset(&mut self) -> bytesmith::Len {
        self.b_end_offset()
    }
    pub fn c_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.c_start_offset().bits()..self.c_end_offset().bits()
    }
    #[allow(dead_code)]
    fn c_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.c_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn c_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}

pub struct Cross<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}
