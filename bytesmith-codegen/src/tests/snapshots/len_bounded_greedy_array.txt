#[allow(non_camel_case_types)]
pub struct Frame_body_Iterator<'a> {
    idx: usize,
    count: usize,
    data: &'a [u8],
}

impl<'a> ::bytesmith::Dissect<'a> for Frame<'a> {
    fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        Frame::field_tree(self)
    }
    fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        Frame::handoff(self)
    }
}

impl<'a> ::std::iter::Iterator for Frame_body_Iterator<'a> {
    type Item = ::bytesmith::ParseResult<u8>;
    fn next(&mut self) -> std::option::Option<Self::Item> {
        if self.idx == self.count {
            return None;
        }
        self.idx += 1;
        let value = self.data[0];
        self.data = &self.data[1..];
        Some(Ok(value))
    }
}

impl<'a> Frame<'a> {
    pub fn parse(data: &'a [u8]) -> Result<(Self, &'a [u8]), ::bytesmith::ParseError> {
        let mut me = Self { data };
        me.tag_fatal_check()?;
        me.tag_recoverable_check()?;
        me.len_fatal_check()?;
        me.len_recoverable_check()?;
        me.body_fatal_check()?;
        me.body_recoverable_check()?;
        me.after_fatal_check()?;
        me.after_recoverable_check()?;
        let len = me.after_end_offset();
        if len.bit != 0 {
            return Err(::bytesmith::ParseError::UnalignedLength(len));
        }
        if data.len() < len.byte {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: len.byte,
                got: data.len(),
            });
        }
        Ok((me, &data[len.byte..]))
    }
    pub fn field_tree(&mut self) -> ::bytesmith::FieldNode<'a> {
        let mut children = ::std::vec::Vec::new();
        {
            let me = &mut *self;
            children.push(me.tag_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.len_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.body_present_node());
        }
        {
            let me = &mut *self;
            children.push(me.after_present_node());
        }
        let mut root = ::bytesmith::FieldNode::new(
                "Frame",
                "Frame",
                0usize..self.after_end_offset().bits(),
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        root.set_paths("");
        root
    }
    pub fn dissect(data: &'a [u8]) -> ::bytesmith::FieldNode<'a> {
        let mut me = Self { data };
        let mut children: ::std::vec::Vec<::bytesmith::FieldNode<'a>> = ::std::vec::Vec::new();
        let mut fatal: Option<::bytesmith::ParseError> = None;
        if fatal.is_none() {
            match me.tag_fatal_check() {
                Err(error) => {
                    let start = me.tag_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "tag",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.tag_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.tag_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "tag",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.tag_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.len_fatal_check() {
                Err(error) => {
                    let start = me.len_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "len",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.len_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.len_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "len",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.len_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.body_fatal_check() {
                Err(error) => {
                    let start = me.body_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "body",
                                    "[u8]",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.body_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.body_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "body",
                                            "[u8]",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.body_present_node());
                        }
                    }
                }
            }
        }
        if fatal.is_none() {
            match me.after_fatal_check() {
                Err(error) => {
                    let start = me.after_start_offset().bits();
                    children
                        .push(
                            ::bytesmith::FieldNode::new(
                                    "after",
                                    "u8",
                                    start..start,
                                    ::bytesmith::Value::Opaque,
                                )
                                .with_status(::bytesmith::Status::Error(error)),
                        );
                    fatal = Some(error);
                }
                Ok(()) => {
                    match me.after_recoverable_check() {
                        Err(error) => {
                            let bit_range = me.after_bit_range();
                            children
                                .push(
                                    ::bytesmith::FieldNode::new(
                                            "after",
                                            "u8",
                                            bit_range,
                                            ::bytesmith::Value::Opaque,
                                        )
                                        .with_status(::bytesmith::Status::Error(error)),
                                );
                        }
                        Ok(()) => {
                            children.push(me.after_present_node());
                        }
                    }
                }
            }
        }
        let root_end = children.last().map(|child| child.bit_range.end).unwrap_or(0);
        let mut root = ::bytesmith::FieldNode::new(
                "Frame",
                "Frame",
                0usize..root_end,
                ::bytesmith::Value::Struct,
            )
            .with_children(children);
        if let Some(error) = fatal {
            root = root.with_status(::bytesmith::Status::Error(error));
        }
        root.set_paths("");
        root
    }
    pub fn handoff(&mut self) -> Option<::bytesmith::Handoff<'a>> {
        None
    }
    #[allow(dead_code)]
    fn tag_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.tag_bit_range();
        ::bytesmith::FieldNode::new(
            "tag",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.tag())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn tag(&mut self) -> u8 {
        self.data[0usize]
    }
    pub fn tag_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
    }
    pub fn tag_start_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len::ZERO
    }
    pub fn tag_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.tag_start_offset().bits()..self.tag_end_offset().bits()
    }
    #[allow(dead_code)]
    fn tag_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.tag_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn tag_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn len_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.len_bit_range();
        ::bytesmith::FieldNode::new(
            "len",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.len())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn len(&mut self) -> u8 {
        self.data[1usize]
    }
    pub fn len_end_offset(&mut self) -> bytesmith::Len {
        bytesmith::Len {
            byte: 2usize,
            bit: 0usize,
        }
    }
    pub fn len_start_offset(&mut self) -> bytesmith::Len {
        self.tag_end_offset()
    }
    pub fn len_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.len_start_offset().bits()..self.len_end_offset().bits()
    }
    #[allow(dead_code)]
    fn len_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.len_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn len_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn body_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.body_bit_range();
        {
            let mut node = {
                let mut elem_nodes = ::std::vec::Vec::new();
                if let Ok(iter) = self.body() {
                    let mut start = bit_range.start;
                    for (i, elem) in iter.enumerate() {
                        let end = start.saturating_add(8usize);
                        match elem {
                            Ok(value) => {
                                elem_nodes
                                    .push(
                                        ::bytesmith::FieldNode::new(
                                            i.to_string(),
                                            "u8",
                                            start..end,
                                            ::bytesmith::Value::UInt(u128::from(value)),
                                        ),
                                    )
                            }
                            Err(error) => {
                                elem_nodes
                                    .push(
                                        ::bytesmith::FieldNode::new(
                                                i.to_string(),
                                                "u8",
                                                start..start,
                                                ::bytesmith::Value::Opaque,
                                            )
                                            .with_status(::bytesmith::Status::Error(error)),
                                    );
                            }
                        }
                        start = end;
                    }
                }
                ::bytesmith::FieldNode::new(
                        "body",
                        "[u8]",
                        bit_range.clone(),
                        ::bytesmith::Value::Array,
                    )
                    .with_children(elem_nodes)
            };
            if let Ok(rest) = self.body_rest() && !rest.is_empty() {
                let consumed = node
                    .children
                    .last()
                    .map(|child| child.bit_range.end)
                    .unwrap_or(bit_range.start)
                    .min(bit_range.end);
                node.children
                    .push(
                        ::bytesmith::FieldNode::new(
                            "rest",
                            "[u8]",
                            consumed..bit_range.end,
                            ::bytesmith::Value::Bytes(rest),
                        ),
                    );
            }
            node
        }
    }
    pub fn body_rest(&mut self) -> ::bytesmith::ParseResult<&'a [u8]> {
        let end = ({ 2usize }).saturating_add(self.len() as usize).min(self.data.len());
        let consumed = (2usize)
            .saturating_add(
                ({
                    {
                        let start = 2usize;
                        ::bytesmith::Len {
                            byte: self
                                .data[..(({ 2usize })
                                    .saturating_add(self.len() as usize)
                                    .min(self.data.len()))]
                                .len()
                                .saturating_sub(start),
                            bit: 0,
                        }
                    }
                })
                    .byte_ceil(),
            );
        if consumed > end {
            return Err(::bytesmith::ParseError::NotEnoughData {
                expected: consumed,
                got: end,
            });
        }
        Ok(&self.data[consumed..end])
    }
    #[allow(clippy::identity_op)]
    pub fn body(&mut self) -> ::bytesmith::ParseResult<Frame_body_Iterator<'a>> {
        Ok(Frame_body_Iterator {
            idx: 0,
            count: self
                .data[..(({ 2usize })
                    .saturating_add(self.len() as usize)
                    .min(self.data.len()))]
                .len()
                .saturating_sub(2usize),
            data: &self
                .data[..(({ 2usize })
                .saturating_add(self.len() as usize)
                .min(self.data.len()))][2usize..],
        })
    }
    pub fn body_end_offset(&mut self) -> bytesmith::Len {
        ::bytesmith::Len {
            byte: 2usize,
            bit: 0usize,
        }
            + ({
                ::bytesmith::Len {
                    byte: self.len() as usize,
                    bit: 0,
                }
            })
    }
    pub fn body_start_offset(&mut self) -> bytesmith::Len {
        self.len_end_offset()
    }
    pub fn body_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.body_start_offset().bits()..self.body_end_offset().bits()
    }
    #[allow(dead_code)]
    fn body_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.body_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn body_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
    #[allow(dead_code)]
    fn after_present_node(&mut self) -> ::bytesmith::FieldNode<'a> {
        let bit_range = self.after_bit_range();
        ::bytesmith::FieldNode::new(
            "after",
            "u8",
            bit_range.clone(),
            ::bytesmith::Value::UInt(u128::from(self.after())),
        )
    }
    #[allow(clippy::identity_op)]
    pub fn after(&mut self) -> u8 {
        {
            let offset = ::bytesmith::Len {
                byte: 2usize,
                bit: 0usize,
            }
                + ({
                    ::bytesmith::Len {
                        byte: self.len() as usize,
                        bit: 0,
                    }
                });
            debug_assert!(offset.bit == 0, "primitive requires byte alignment");
            self.data[offset.byte]
        }
    }
    pub fn after_end_offset(&mut self) -> bytesmith::Len {
        ::bytesmith::Len {
            byte: 1usize,
            bit: 0usize,
        }
            + ({
                ::bytesmith::Len {
                    byte: 2usize,
                    bit: 0usize,
                }
                    + ({
                        ::bytesmith::Len {
                            byte: self.len() as usize,
                            bit: 0,
                        }
                    })
            })
    }
    pub fn after_start_offset(&mut self) -> bytesmith::Len {
        self.body_end_offset()
    }
    pub fn after_bit_range(&mut self) -> ::core::ops::Range<usize> {
        self.after_start_offset().bits()..self.after_end_offset().bits()
    }
    #[allow(dead_code)]
    fn after_fatal_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        {
            let len = self.after_end_offset();
            let expected = len.byte_ceil();
            if self.data.len() < expected {
                return Err(::bytesmith::ParseError::NotEnoughData {
                    expected,
                    got: self.data.len(),
                });
            }
        }
        Ok(())
    }
    #[allow(dead_code)]
    fn after_recoverable_check(&mut self) -> Result<(), ::bytesmith::ParseError> {
        Ok(())
    }
}

pub struct Frame<'a> {
    #[allow(dead_code)]
    data: &'a [u8],
}
